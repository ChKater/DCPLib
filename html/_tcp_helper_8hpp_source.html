54  void prepareRead() {
55  asio::async_read(*socket,
56  asio::buffer(data, maxLength),
57  std::bind(&Session::completion_condition, this,
58  std::placeholders::_1,
59  std::placeholders::_2),

61  shared_from_this(),
62  std::placeholders::_1,
63  std::placeholders::_2));

66  std::size_t completion_condition(const std::error_code &error, std::size_t bytes_transferred) {
67  if (!error && bytes_transferred >= 4) {
68  return (*((uint32_t *) data) + 4) - bytes_transferred;
69  }
70  return 4 - bytes_transferred;
71  }
72 
73  void handleRead(std::shared_ptr<Session> s, const std::error_code &error, size_t bytes_transferred) {
74  if (!error) {
75  if (sessionManager != nullptr) {
76  sessionManager->setLastSessionAccess(id);
77  }
78  DcpPdu *pdu = makeDcpPdu(data, bytes_transferred - 4);
79 #if defined(DEBUG)
80  Log(PDU_RECEIVED, pdu->to_string());
81 #endif
82  dcpManager.receive(*pdu);
83  delete pdu;
84 
85  prepareRead();
86 
87 
88  } else {
89  if (asio::error::connection_reset == error || asio::error::operation_aborted == error ||
90  asio::error::eof == error) {
91  //Session is closed => stop receiving
92  if (sessionManager != nullptr) {
93  sessionManager->removeSession(id);
94  }
95  return;
96  }
97 
98  dcpManager.reportError(DcpError::PROTOCOL_ERROR_GENERIC);
99 #if defined(DEBUG) || defined(LOGGING)
100  Log(NETWORK_PROBLEM, Tcp::protocolName, error.message());
101 #endif
102  if (sessionManager != nullptr) {
103  sessionManager->removeSession(id);
104  }
105  return;
106 
107  }
108  }

114  std::error_code error;

116  std::size_t size = socket->send(

120 #if defined(DEBUG) || defined(LOGGING)
