46  virtual void receive(DcpPdu &msg) override {
47  switch (msg.getTypeId()) {
48  case DcpPduType::RSP_ack: {
49  DcpPduAck &ack = static_cast<DcpPduAck &>(msg);
50  if (synchronousCallback[DcpCallbackTypes::ACK]) {
51  ackReceivedListener(ack.getSender(), ack.getRespSeqId());
52  } else {
53  std::thread t(ackReceivedListener, ack.getSender(), ack.getRespSeqId());
54  t.detach();
55  }
56  break;
57  }
58  case DcpPduType::RSP_nack: {
59  DcpPduNack &nack = static_cast<DcpPduNack &>(msg);
60  if (synchronousCallback[DcpCallbackTypes::NACK]) {
61  nAckReceivedListener(nack.getSender(), nack.getRespSeqId(),
62  nack.getErrorCode());
63  } else {
64  std::thread t(nAckReceivedListener, nack.getSender(), nack.getRespSeqId(),
65  nack.getErrorCode());
66  t.detach();
67  }
68  break;
69  }
70  case DcpPduType::RSP_state_ack: {
71  DcpPduStateAck &stateAck = static_cast<DcpPduStateAck &>(msg);
72  if (synchronousCallback[DcpCallbackTypes::NACK]) {
73  stateAckReceivedListener(stateAck.getSender(),
74  stateAck.getRespSeqId(), stateAck.getStateId());
75  } else {
76  std::thread t(stateAckReceivedListener, stateAck.getSender(),
77  stateAck.getRespSeqId(), stateAck.getStateId());
78  t.detach();
79  }
80  break;
81  }
82  case DcpPduType::RSP_error_ack: {
83  DcpPduNack &errorAck = static_cast<DcpPduNack &>(msg);
84  DcpPduStateAck &stateAck = static_cast<DcpPduStateAck &>(msg);
85  if (synchronousCallback[DcpCallbackTypes::NACK]) {
86  errorAckReceivedListener(errorAck.getSender(),
87  errorAck.getRespSeqId(), errorAck.getErrorCode());
88  } else {
89  std::thread t(errorAckReceivedListener, errorAck.getSender(),
90  errorAck.getRespSeqId(), errorAck.getErrorCode());
91  t.detach();
92  }
93  break;
94  }
95  case DcpPduType::RSP_log_ack: {
96  DcpPduLogAck &logAck = static_cast<DcpPduLogAck &>(msg);
97  if (logTemplates.count(logAck.getSender()) > 0) {
98 
99  std::vector<std::shared_ptr<LogEntry>> entries;
100  size_t offset = 0;
101  while (offset < logAck.getSerializedSize() - 4) {
102  uint8_t *curLog = logAck.getPayload() + offset;
103  uint8_t templateId = *((uint8_t *) curLog + 8);
104  if (logTemplates[logAck.getSender()].count(templateId) > 0) {
105 
106  LogTemplate &logTemplate = logTemplates[logAck.getSender()].find(templateId)->second;
107  LogEntry *logEntry = new LogEntry(logTemplate, curLog, 0);
108  logEntry->applyPayloadToString();
109  uint8_t *payload = new uint8_t[logEntry->serializedSize()];
110  memcpy(payload, curLog, logEntry->serializedSize());
111  logEntry->setPayload(payload);
112 
113  entries.push_back(std::shared_ptr<LogEntry>(logEntry));
114  offset += logEntry->serializedSize();
115 
116  } else {
117  break;
118  //toDo log unknown id
119  }
120  }
121  if (synchronousCallback[DcpCallbackTypes::RSP_log_ack]) {
122  logAckListener(logAck.getSender(), logAck.getRespSeqId(), entries);
123  } else {
124  std::thread t(logAckListener, logAck.getSender(), logAck.getRespSeqId(), entries);
125  t.detach();
126  }
127  }
128  break;
129  }
130  case DcpPduType::NTF_state_changed: {
131  DcpPduNtfStateChanged &stateChanged =
132  static_cast<DcpPduNtfStateChanged &>(msg);
133  if (synchronousCallback[DcpCallbackTypes::NACK]) {
134  stateChangedNotificationReceivedListener(stateChanged.getSender(),
135  stateChanged.getStateId());
136  } else {
137  std::thread t(stateChangedNotificationReceivedListener, stateChanged.getSender(),
138  stateChanged.getStateId());
139  t.detach();
140  }
141  break;
142  }
143  case DcpPduType::NTF_log: {
144  DcpPduNtfLog &log = static_cast<DcpPduNtfLog &>(msg);
145  if (logTemplates.count(log.getSender()) > 0 &&
146  logTemplates[log.getSender()].count(log.getTemplateId()) > 0) {
147  uint8_t templateId = *((uint8_t *) log.getPayload() + 10);
148  if (logTemplates[log.getSender()].count(templateId) > 0) {
149 
150  size_t size = log.getSerializedSize() - 2;
151  uint8_t *payload = new uint8_t[size];
152  memcpy(payload, log.getLogEntryPayload(), size);
153  LogTemplate &logTemplate = logTemplates[log.getSender()].find(templateId)->second;
154  const std::shared_ptr<LogEntry> sharedPtr =
155  std::shared_ptr<LogEntry>(new LogEntry(logTemplate, payload,
156  size));
157 
158  if (synchronousCallback[DcpCallbackTypes::NTF_LOG]) {
159  logNotificationListener(log.getSender(), sharedPtr);
160  } else {
161  std::thread t(logNotificationListener, log.getSender(), sharedPtr);
162  t.detach();
163  }
164  } else {
165  break;
166  //toDo log unknown id
167  }
168  }
169  break;
170  }
171  case DcpPduType::DAT_input_output: {
172  DcpPduDatInputOutput &data = static_cast<DcpPduDatInputOutput &>(msg);
173  if (synchronousCallback[DcpCallbackTypes::NACK]) {
174  dataReceivedListener(data.getDataId(), data.getSerializedSize(), data.getPayload());
175  } else {
176  std::thread t(dataReceivedListener, data.getDataId(), data.getSerializedSize(), data.getPayload());
177  t.detach();
178  }
179  break;
180  }
181  }
182  pduListener(msg);
183  }

190  t.detach();
191  }

643  void enableHeartbeat(const uint8_t dcpId, const uint32_t numerator, const uint32_t denominator) {
644  //close existing heartbeat thread if exist
645  disableHeartbeat(dcpId);
646  runningHeartbeats[dcpId] = true;
647  heartbeatThreads[dcpId] = std::unique_ptr<std::thread>(
648  new std::thread(&DcpManagerMaster::heartBeatRoutine, this, dcpId, numerator, denominator));
649  }
650 
657  void disableHeartbeat(const uint8_t dcpId) {
658  if (heartbeatThreads.count(dcpId)) {
659  runningHeartbeats[dcpId] = false;
660  heartbeatCV[dcpId].notify_all();
661  heartbeatThreads[dcpId]->join();
662  heartbeatThreads.erase(dcpId);
663  runningHeartbeats.erase(dcpId);
664  }
665  }

672  void setDcpSlaveDescription(const uint8_t dcpId, const SlaveDescription_t &slaveDescription) {
673  if (slaveDescription.Log.get() != nullptr) {
674  for (const auto &templateObj : slaveDescription.Log->templates) {
675  std::vector<DcpDataType> cDataTypes;
676  std::vector<std::string> strs;
677  std::string buf; // Have a buffer string
678  std::stringstream ss(templateObj.msg);
679  while (ss >> buf)
680  strs.push_back(buf);
681  for (const std::string &str: strs) {
682  if (str[0] == '%') {
683  cDataTypes.push_back(from_string_DcpDataType(str.substr(1)));
684  }
685  }
686  logTemplates[dcpId].insert(std::make_pair(templateObj.id,
687  LogTemplate(templateObj.id, templateObj.category,
688  (DcpLogLevel) templateObj.level, templateObj.msg,
689  cDataTypes)));
690 
691  }
692  }
